93 void MainView::paintGL() {
94 
95  glClearBufferfv(GL_COLOR, 0, glm::value_ptr(MainModel::skyColor));
96  glClear(GL_DEPTH_BUFFER_BIT);
97  glViewport(0, 0, mainViewWidth_, mainViewHeight_);
98  glEnable(GL_CULL_FACE);
99  glEnable(GL_CLIP_PLANE0);
100 
101  cameraOn_ = !(cameras_->empty());
102 
103  V_ = view_->getViewMatrix();
104  MV_ = V_ * M_;
105  MVP_ = P_ * MV_;
106 
107  glBindVertexArray(vaoMainViewID_);
108  shaderProgram_.use();
109 
110  static int numOfCams;
111  numOfCams = cameras_->size();
112 
113  glUniform1i(shaderProgram_("numOfCams"), numOfCams);
114  glUniformMatrix4fv(shaderProgram_("MV"), 1, GL_FALSE, glm::value_ptr(MV_));
115  glUniformMatrix4fv(shaderProgram_("MVP"), 1, GL_FALSE, glm::value_ptr(MVP_));
116 
117  if (cameraOn_) {
118  static glm::mat4 projectorMatrix[7] = {glm::mat4(0.0f)};
119 
120 
121  //Loads all exisiting camera projector matrices and their corresponding
122  //depth textures (z-buffers) for use in MainView's shader
123  int cameraCount = 0;
124  for (auto itrCamera = cameras_->cbegin(); itrCamera != cameras_->cend(); ++itrCamera) {
125  projectorMatrix[cameraCount] = GLUtils::coordSpaceToTextureSamplingAdapter *
126  cameras_->getCameraViewProjectorMatrix(itrCamera.key());
127 
128  glActiveTexture(GL_TEXTURE1 + cameraCount++);
129  glBindTexture(GL_TEXTURE_2D, cameras_->getCameraViewZBuffer(itrCamera.key()));
130  }
131 
132  glUniformMatrix4fv(shaderProgram_("projectionMat"), numOfCams, GL_FALSE, glm::value_ptr(projectorMatrix[0]));
133 
134  //If feature is enabled, loads 180 FOV camera projector matrix and its corresponding
135  //depth texture (z-buffer) for use in MainView's shader
136  if (camera180FOVView_) {
137  static glm::mat4 projectorPotentialMatrix = glm::mat4(0.0f);
138  projectorPotentialMatrix = GLUtils::coordSpaceToTextureSamplingAdapter *
139  cameras_->getSelectedCamera180FOVProjectorMatrix();
140 
141  glUniform1i(shaderProgram_("is180FOVOn"), 1);
142  glUniformMatrix4fv(shaderProgram_("potentialProjMat"), 1, GL_FALSE, glm::value_ptr(projectorPotentialMatrix));
143  glActiveTexture(GL_TEXTURE8);
144  glBindTexture(GL_TEXTURE_2D, cameras_->getSelectedCamera180FOVZBuffer());
145  } else {
146  glUniform1i(shaderProgram_("is180FOVOn"), 0);
147  }
148  }
149 
150  mainModel_.drawModel(shaderProgram_);
151 
152  shaderProgram_.unUse();
153 
154  if (cameraOn_) {
155  cameras_->drawCameras(MVP_);
156  cameras_->drawSelectedCameraFrustum(MVP_);
157  cameras_->drawCameraLabels(V_, P_);
158  }
159 
160  if (changeScene_ == true)
161  changeScene_ = view_->gotoNewPosition();
162 
163  renderLater();
164 }

244 glm::vec3 MainView::getMouseCursorWolrdPosition(const float winPosX, const float winPosY) {
245  GLint viewport[4];
246  glGetIntegerv(GL_VIEWPORT, viewport);
247 
248  //point 0 y-coord in opengl starts at the bottom, whereas in this window at the top
249  //when calling glFunctions the y must be converted to suit opengl requirements
250  float SCREEN_SPACE_Y = viewport[3] - winPosY;
251  float SCREEN_SPACE_X = winPosX;
252 
253  GLfloat screenSpaceZ;
254  glReadPixels(SCREEN_SPACE_X, SCREEN_SPACE_Y, 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &screenSpaceZ);
255 
256  glm::vec3 screenSpacePos = glm::vec3(SCREEN_SPACE_X, SCREEN_SPACE_Y, screenSpaceZ);
257 
258  glm::vec3 worldSpacePos = glm::unProject(screenSpacePos, view_->getViewMatrix(),
259  P_, glm::vec4(viewport[0], viewport[1], viewport[2], viewport[3]));
260 
261  return worldSpacePos;
262 }
