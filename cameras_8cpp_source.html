144 void Cameras::getMemento() {
145  if (!mementoStack_.empty()) {
146 
147  QMap<int, Camera> newCameras = cameras_;
148  cameras_ = std::move(mementoStack_.top().getState()); //store previous state in cameras
149 
150  if (newCameras.size() == cameras_.size()) { //camera not added/deleted
151  setSelectedCamera(selectedCamID_);
152  }
153  else if (newCameras.size() > cameras_.size()) { //delete camera to complete undo
154  deleteCameraResources(getCamIDForUndoOperation(newCameras));
155  }
156  else { //add camera to complete undo
157  addCameraResources(getCamIDForUndoOperation(newCameras));
158  }
159  drawAllCameraZBuffers();
160  mementoStack_.pop();
161  }
162 }

220 int Cameras::getCamIDForUndoOperation(QMap<int, Camera> &newState) {
